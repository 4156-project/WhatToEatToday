public class BaseResponse implements Serializable{

    private StatusCode code; //
    private String message; //
    private QueryTableResponse data; //

    public StatusCode getCode(){
        return code;
    }

    public void setCode(StatusCode code){
        this.code = code;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public QueryTableResponse getData(){
        return data;
    }

    public void setData(QueryTableResponse data){
        this.data = data;
    }

}

public class QueryTableResponse implements Serializable{

    private String tableName; //
    private Object filedNameTypeMap; //
    private long rowNum; //
    private int id; //

    public String getTableName(){
        return tableName;
    }

    public void setTableName(String tableName){
        this.tableName = tableName;
    }

    public Object getFiledNameTypeMap(){
        return filedNameTypeMap;
    }

    public void setFiledNameTypeMap(Object filedNameTypeMap){
        this.filedNameTypeMap = filedNameTypeMap;
    }

    public long getRowNum(){
        return rowNum;
    }

    public void setRowNum(long rowNum){
        this.rowNum = rowNum;
    }

    public int getId(){
        return id;
    }

    public void setId(int id){
        this.id = id;
    }

}
